1. Create src/actions/schedule.ts following the exact pattern from beauty-products.ts:
   - "use server" directive at top
   - Import createClient from "@/lib/supabase/server", db from "@/lib/db", scheduleJobs from "@/lib/db/schema"
   - Import eq, and, gte, lte, desc, sql from "drizzle-orm"
   - Import z from "zod"

2. Define zod schemas:
   - createJobSchema: jobDate as string regex /^\d{4}-\d{2}-\d{2}$/, clientName string min(1) max(100), location string min(1) max(200), startTime string regex /^\d{2}:\d{2}$/, endTime string regex /^\d{2}:\d{2}$/, payAmount number int min(0), status z.enum(["paid", "pending"]), notes string max(1000) optional nullable
   - updateJobSchema: same fields but all optional except at least one must be provided

3. Implement server actions (each starts with auth gate):
   - createJob(data): validate with createJobSchema, insert into scheduleJobs, return created record
   - getJobsForMonth(year: number, month: number): query where jobDate >= "YYYY-MM-01" and jobDate <= "YYYY-MM-31", order by jobDate asc then startTime asc
   - getJobById(jobId: string): validate UUID, select where id equals, return single job or null
   - updateJob(jobId: string, data): validate UUID and updateJobSchema, update with set({ ...parsed.data, updatedAt: new Date() }), return updated record
   - deleteJob(jobId: string): validate UUID, delete where id equals, return { success: true }
   - getIncomeStats(year: number, month: number): fetch all jobs for month, compute totalPaid (sum of payAmount where status=paid), totalPending (sum where status=pending), total = totalPaid + totalPending. Return { totalPaid, totalPending, total, jobCount }
   - getYearlyStats(year: number): fetch all jobs for year (jobDate >= "YYYY-01-01" and <= "YYYY-12-31"), group by month, return array of { month, totalPaid, totalPending, total }

4. Create src/__tests__/schedule/schedule-actions.test.ts:
