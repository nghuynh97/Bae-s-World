   import { relations } from "drizzle-orm";

   export const categoriesRelations = relations(categories, ({ many }) => ({
     portfolioItems: many(portfolioItems),
   }));

   export const portfolioItemsRelations = relations(portfolioItems, ({ one }) => ({

1. **Create `src/actions/portfolio.ts`** with these Server Actions:

   **`getPortfolioItems(cursor?: string, categorySlug?: string)`** -- PUBLIC (no auth required):
   - PAGE_SIZE = 12
   - Select from portfolioItems joined with images, imageVariants, and categories
   - If categorySlug provided and not "all", filter by categories.slug
   - If cursor provided, filter where portfolioItems.createdAt < new Date(cursor)
   - Order by portfolioItems.createdAt DESC
   - Limit PAGE_SIZE + 1, check hasMore, return { items, nextCursor }
   - Each item should include: portfolioItem fields, category name/slug, image width/height, and a variants array with { variantName, storagePath, width, height }
   - Construct public image URLs using getPublicImageUrl("public-images", variant.storagePath) for each variant

   **`createPortfolioItem(data: { title: string, description?: string, categoryId: string, imageId: string })`** -- AUTH REQUIRED:
   - Validate with zod: title min 1 max 100, description max 500, categoryId uuid, imageId uuid
   - Auth check: `const { data: { user } } = await supabase.auth.getUser(); if (!user) throw new Error("Unauthorized");`
   - Both users can create (no role check beyond auth -- this satisfies AUTH-06)
   - Insert into portfolioItems, return the created item

   **`updatePortfolioItem(itemId: string, data: { title?: string, description?: string, categoryId?: string })`** -- AUTH REQUIRED:
   - Validate with zod
   - Auth check
   - Update portfolioItems where id = itemId, set updatedAt to new Date()
   - Return updated item

   **`deletePortfolioItem(itemId: string)`** -- AUTH REQUIRED:
   - Auth check
   - Find the portfolio item to get its imageId
   - Delete all image variants from Supabase Storage (get storagePaths from imageVariants table, then supabase.storage.from("public-images").remove(paths))
   - Delete the image record (cascades to imageVariants via onDelete: cascade)
   - Delete the portfolioItem record
   - Return { success: true }
